1. DANS UN COMPOSANT REACT

import { useHumanBehavior } from '@/hooks/useHumanBehavior';
import { ConfidenceIndicator } from '@/components/ConfidenceIndicator';

function ChatMessage({ message, query }) {
  const { enrichResponse, checkHallucination } = useHumanBehavior();
  
  const enriched = enrichResponse(message, query);
  const halluCheck = checkHallucination(message);
  
  return (
    <div>
      {enriched.thinking && <p>{enriched.thinking}</p>}
      <p>{enriched.response}</p>
      <ConfidenceIndicator 
        confidence={enriched.confidence}
        hallucinationWarnings={halluCheck.warnings}
      />
    </div>
  );
}

2. DÉTECTION D'HALLUCINATION

import { detectPotentialHallucination } from '@/utils/humanBehavior';

const check = detectPotentialHallucination(response);
if (check.likely) {
  alert("⚠️ Hallucination détectée : " + check.warnings[0]);
}

const validation = validateAgentConsensus(agentResponses);

🔧 MODIFICATIONS DES AGENTS
═══════════════════════════════════════════════════════════════════════════════

Tous les agents ont été enrichis avec des instructions de comportement humain :

✅ AGENT LOGIQUE
   + Si information manquante → ADMETS-LE
   + N'invente JAMAIS de données
   + Calibre certitude : "certain", "probable", "hypothétique"

✅ AGENT CRÉATIF
   + Distingue "idée spéculative" vs "proposition concrète"
   + Précise le sens des analogies
   + Créatif SANS inventer de faux faits

✅ AGENT CRITIQUE
   + Distingue "risque prouvé" vs "risque hypothétique"
   + Si erreur, CITE pourquoi
   + Reconnais les arguments solides

✅ AGENT SYNTHÉTISEUR
   + Utilise indicateurs : 🟢 🟡 🔴
   + Si info clé manque → DIS-LE
   + Si hallucination → ALERTE
